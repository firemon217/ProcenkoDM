def fib_naive(n):
    # –ë–∞–∑–æ–≤—ã–µ —Å–ª—É—á–∞–∏: F(0)=0, F(1)=1

def fib_memo(n, memo=None):
    if memo is None:
        memo = {}

    if n <= 1:
        return n

    # –ï—Å–ª–∏ –∑–Ω–∞—á–µ–Ω–∏–µ —É–∂–µ –≤—ã—á–∏—Å–ª—è–ª–æ—Å—å ‚Äî –≤–µ—Ä–Ω—É—Ç—å –∏–∑ memo
    if n in memo:
        return memo[n]

    # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç, —á—Ç–æ–±—ã –Ω–µ –≤—ã—á–∏—Å–ª—è—Ç—å –ø–æ–≤—Ç–æ—Ä–Ω–æ

def fib_iter(n):
    if n <= 1:
        return n

    # –°–æ–∑–¥–∞—ë–º —Ç–∞–±–ª–∏—Ü—É dp[0..n]
    dp = [0] * (n + 1)
    dp[1] = 1

    # –ó–∞–ø–æ–ª–Ω—è–µ–º —Ç–∞–±–ª–∏—Ü—É —Å–Ω–∏–∑—É –≤–≤–µ—Ä—Ö
    for i in range(2, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2]

    return dp[n]

def knapsack(weights, values, capacity):
    n = len(weights)

    # dp[i][w] ‚Äî –º–∞–∫—Å–∏–º—É–º —Ü–µ–Ω–Ω–æ—Å—Ç–∏, –µ—Å–ª–∏ –±–µ—Ä–µ–º –ø–µ—Ä–≤—ã–µ i –ø—Ä–µ–¥–º–µ—Ç–æ–≤ –∏ —Ä—é–∫–∑–∞–∫ –∏–º–µ–µ—Ç –≤–º–µ—Å—Ç–∏–º–æ—Å—Ç—å w
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]

    # –ó–∞–ø–æ–ª–Ω—è–µ–º —Ç–∞–±–ª–∏—Ü—É
    for i in range(1, n + 1):
        w = weights[i - 1]  # –≤–µ—Å —Ç–µ–∫—É—â–µ–≥–æ –ø—Ä–µ–¥–º–µ—Ç–∞
        v = values[i - 1]   # —Ü–µ–Ω–Ω–æ—Å—Ç—å —Ç–µ–∫—É—â–µ–≥–æ –ø—Ä–µ–¥–º–µ—Ç–∞

        for cap in range(1, capacity + 1):

            if w > cap:
                # –ù–µ –º–æ–∂–µ–º –≤–∑—è—Ç—å –ø—Ä–µ–¥–º–µ—Ç ‚Äî –ø—Ä–æ—Å—Ç–æ –∫–æ–ø–∏—Ä—É–µ–º –∑–Ω–∞—á–µ–Ω–∏–µ —Å–≤–µ—Ä—Ö—É
                dp[i][cap] = dp[i - 1][cap]
            else:
                # –ú–∞–∫—Å–∏–º—É–º –º–µ–∂–¥—É –∏–≥–Ω–æ—Ä–∏—Ä–æ–≤–∞–Ω–∏–µ–º –∏ –≤–∫–ª—é—á–µ–Ω–∏–µ–º –ø—Ä–µ–¥–º–µ—Ç–∞
                dp[i][cap] = max(
                    dp[i - 1][cap],            # –Ω–µ –±–µ—Ä–µ–º –ø—Ä–µ–¥–º–µ—Ç
                    dp[i - 1][cap - w] + v     # –±–µ—Ä–µ–º –ø—Ä–µ–¥–º–µ—Ç
                )

    return dp[n][capacity]
# –í—Ä–µ–º—è: O(n * W)
# –ü–∞–º—è—Ç—å: O(n * W)

def lcs(a, b):
    n, m = len(a), len(b)

    # dp[i][j] ‚Äî –¥–ª–∏–Ω–∞ LCS –¥–ª—è a[:i] –∏ b[:j]
    dp = [[0] * (m + 1) for _ in range(n + 1)]

    # –ó–∞–ø–æ–ª–Ω—è–µ–º —Ç–∞–±–ª–∏—Ü—É
    for i in range(1, n + 1):
        for j in range(1, m + 1):

            if a[i - 1] == b[j - 1]:
                # –°–∏–º–≤–æ–ª—ã —Å–æ–≤–ø–∞–¥–∞—é—Ç ‚Üí —É–≤–µ–ª–∏—á–∏–≤–∞–µ–º LCS –Ω–∞ 1
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                # –ò–Ω–∞—á–µ –±–µ—Ä—ë–º –º–∞–∫—Å–∏–º—É–º –∏–∑ –≤–µ—Ä—Ö–Ω–µ–≥–æ –∏ –ª–µ–≤–æ–≥–æ
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    return dp[n][m]
# –í—Ä–µ–º—è: O(n * m)
# –ü–∞–º—è—Ç—å: O(n * m)


def levenshtein(a, b):
    n, m = len(a), len(b)

    # dp[i][j] ‚Äî –º–∏–Ω–∏–º–∞–ª—å–Ω–∞—è —Å—Ç–æ–∏–º–æ—Å—Ç—å –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è –ø–µ—Ä–≤—ã—Ö i —Å–∏–º–≤–æ–ª–æ–≤ a –≤ –ø–µ—Ä–≤—ã–µ j —Å–∏–º–≤–æ–ª–æ–≤ b
    dp = [[0] * (m + 1) for _ in range(n + 1)]

    # –ï—Å–ª–∏ —Å—Ç—Ä–æ–∫–∞ –ø—É—Å—Ç–∞—è ‚Äî —Å—Ç–æ–∏–º–æ—Å—Ç—å —Ä–∞–≤–Ω–∞ –∫–æ–ª–∏—á–µ—Å—Ç–≤—É –≤—Å—Ç–∞–≤–æ–∫/—É–¥–∞–ª–µ–Ω–∏–π
    for i in range(n + 1):
        dp[i][0] = i
    for j in range(m + 1):
        dp[0][j] = j

    # –ó–∞–ø–æ–ª–Ω—è–µ–º —Ç–∞–±–ª–∏—Ü—É
    for i in range(1, n + 1):
        for j in range(1, m + 1):

            if a[i - 1] == b[j - 1]:
                # –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª—ã —Ä–∞–≤–Ω—ã ‚Äî —Å—Ç–æ–∏–º–æ—Å—Ç—å –Ω–µ —É–≤–µ–ª–∏—á–∏–≤–∞–µ—Ç—Å—è
                dp[i][j] = dp[i - 1][j - 1]
            else:
                # –í—ã–±–∏—Ä–∞–µ–º –º–∏–Ω–∏–º–∞–ª—å–Ω—É—é –æ–ø–µ—Ä–∞—Ü–∏—é: —É–¥–∞–ª–µ–Ω–∏–µ, –≤—Å—Ç–∞–≤–∫–∞ –∏–ª–∏ –∑–∞–º–µ–Ω–∞
                dp[i][j] = 1 + min(
                    dp[i - 1][j],    # —É–¥–∞–ª–µ–Ω–∏–µ
                    dp[i][j - 1],    # –≤—Å—Ç–∞–≤–∫–∞
                    dp[i - 1][j - 1] # –∑–∞–º–µ–Ω–∞
                )

    return dp[n][m]
# –í—Ä–µ–º—è: O(n * m)
# –ü–∞–º—è—Ç—å: O(n * m)

def lcs_with_reconstruction(a, b):
    n, m = len(a), len(b)

    # –°—Ç—Ä–æ–∏–º DP-—Ç–∞–±–ª–∏—Ü—É
    dp = [[0] * (m + 1) for _ in range(n + 1)]

    for i in range(1, n + 1):
        for j in range(1, m + 1):
            if a[i - 1] == b[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # –¢–µ–ø–µ—Ä—å –≤–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º LCS
    i, j = n, m
    lcs_str = []

    while i > 0 and j > 0:
        # –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª—ã —Å–æ–≤–ø–∞–¥–∞—é—Ç - –¥–æ–±–∞–≤–ª—è–µ–º –≤ –æ—Ç–≤–µ—Ç
        if a[i - 1] == b[j - 1]:
            lcs_str.append(a[i - 1])
            i -= 1
            j -= 1
        else:
            # –ò–¥–µ–º —Ç—É–¥–∞, –≥–¥–µ –±–æ–ª—å—à–µ –∑–Ω–∞—á–µ–Ω–∏–µ
            if dp[i - 1][j] >= dp[i][j - 1]:
                i -= 1
            else:
                j -= 1

    # –†–∞–∑–≤–æ—Ä–∞—á–∏–≤–∞–µ–º, —Ç–∞–∫ –∫–∞–∫ –≤–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–ª–∏ —Å –∫–æ–Ω—Ü–∞
    lcs_str.reverse()

    return dp[n][m], "".join(lcs_str)

def knapsack_with_items(weights, values, capacity):
    n = len(weights)

    # –§–æ—Ä–º–∏—Ä—É–µ–º DP-—Ç–∞–±–ª–∏—Ü—É
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]

    for i in range(1, n + 1):
        w = weights[i - 1]
        v = values[i - 1]

        for cap in range(1, capacity + 1):
            if w > cap:
                dp[i][cap] = dp[i - 1][cap]
            else:
                dp[i][cap] = max(
                    dp[i - 1][cap],
                    dp[i - 1][cap - w] + v
                )

    # –í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ –ø—Ä–µ–¥–º–µ—Ç–æ–≤
    result_value = dp[n][capacity]
    items = []

    cap = capacity
    i = n

    while i > 0:
        # –ï—Å–ª–∏ –∑–Ω–∞—á–µ–Ω–∏–µ –æ—Ç–ª–∏—á–∞–µ—Ç—Å—è –æ—Ç –≤–µ—Ä—Ö–Ω–µ–π —Å—Ç—Ä–æ–∫–∏ ‚Äî –ø—Ä–µ–¥–º–µ—Ç –±—ã–ª –≤–∑—è—Ç
        if dp[i][cap] != dp[i - 1][cap]:
            items.append(i - 1)  # —Å–æ—Ö—Ä–∞–Ω—è–µ–º –∏–Ω–¥–µ–∫—Å –ø—Ä–µ–¥–º–µ—Ç–∞
            cap -= weights[i - 1]
        i -= 1

    items.reverse()
    return result_value, items

def coin_change_min_coins(coins, target_sum):
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –º–æ–Ω–µ—Ç –¥–ª—è —Å—É–º–º—ã target_sum.
    –ï—Å–ª–∏ —Å—É–º–º—É —Å–æ–±—Ä–∞—Ç—å –Ω–µ–ª—å–∑—è ‚Äî –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç -1.

    coins: —Å–ø–∏—Å–æ–∫ –Ω–æ–º–∏–Ω–∞–ª–æ–≤ –º–æ–Ω–µ—Ç
    target_sum: —Ü–µ–ª–µ–≤–∞—è —Å—É–º–º–∞
    """

    # dp[x] —Ö—Ä–∞–Ω–∏—Ç –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –º–æ–Ω–µ—Ç –¥–ª—è —Å—É–º–º—ã x
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º "–±–µ—Å–∫–æ–Ω–µ—á–Ω–æ—Å—Ç—å—é" ‚Äî –±–æ–ª—å—à–æ–µ —á–∏—Å–ª–æ,
    # —á—Ç–æ–±—ã –º–æ–∂–Ω–æ –±—ã–ª–æ –∏—Å–∫–∞—Ç—å –º–∏–Ω–∏–º—É–º.
    dp = [float('inf')] * (target_sum + 1)

    # –°—É–º–º–∞ 0 –Ω–∞–±–∏—Ä–∞–µ—Ç—Å—è –Ω—É–ª—ë–º –º–æ–Ω–µ—Ç
    dp[0] = 0

    # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –∫–∞–∂–¥—É—é —Å—É–º–º—É –æ—Ç 1 –¥–æ target_sum
    for s in range(1, target_sum + 1):

        # –ü—Ä–æ–±—É–µ–º –≤–∑—è—Ç—å –∫–∞–∂–¥—É—é –º–æ–Ω–µ—Ç—É
        for coin in coins:
            if s - coin >= 0:
                dp[s] = min(dp[s], dp[s - coin] + 1)

    # –ï—Å–ª–∏ dp[target_sum] –æ—Å—Ç–∞–ª–æ—Å—å –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ—Å—Ç—å—é ‚Äî —Å—É–º–º–∞ –Ω–µ–¥–æ—Å—Ç–∏–∂–∏–º–∞
    return dp[target_sum] if dp[target_sum] != float('inf') else -1

def lis_dp(arr):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç –¥–ª–∏–Ω—É LIS (–Ω–∞–∏–±–æ–ª—å—à–µ–π –≤–æ–∑—Ä–∞—Å—Ç–∞—é—â–µ–π –ø–æ–¥–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏)
    —Å –ø–æ–º–æ—â—å—é –∫–ª–∞—Å—Å–∏—á–µ—Å–∫–æ–≥–æ –¥–∏–Ω–∞–º–∏—á–µ—Å–∫–æ–≥–æ –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è O(n¬≤).

    arr: –≤—Ö–æ–¥–Ω–∞—è –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å —á–∏—Å–µ–ª
    """

    n = len(arr)
    if n == 0:
        return 0

    # dp[i] ‚Äî –¥–ª–∏–Ω–∞ LIS, –∑–∞–∫–∞–Ω—á–∏–≤–∞—é—â–µ–π—Å—è –≤ –ø–æ–∑–∏—Ü–∏–∏ i
    dp = [1] * n  # –º–∏–Ω–∏–º–∞–ª—å–Ω–∞—è LIS —Å–æ—Å—Ç–æ–∏—Ç —Ö–æ—Ç—è –±—ã –∏–∑ –æ–¥–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞

    # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –∫–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç –∫–∞–∫ –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω—ã–π –∫–æ–Ω–µ—Ü –ø–æ–¥–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
    for i in range(n):

        # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ –ø—Ä–µ–¥—ã–¥—É—â–∏–µ —ç–ª–µ–º–µ–Ω—Ç—ã j < i
        for j in range(i):
            # –ï—Å–ª–∏ arr[j] < arr[i], —Ç–æ arr[i] –º–æ–∂–Ω–æ –¥–æ–±–∞–≤–∏—Ç—å –ø–æ—Å–ª–µ arr[j]
            if arr[j] < arr[i]:
                # –û–±–Ω–æ–≤–ª—è–µ–º dp[i] ‚Äî –±–µ—Ä–µ–º –º–∞–∫—Å–∏–º—É–º –∏–∑ –≤—Å–µ—Ö –≤–∞—Ä–∏–∞–Ω—Ç–æ–≤
                dp[i] = max(dp[i], dp[j] + 1)

    # –û—Ç–≤–µ—Ç ‚Äî –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ dp
    return max(dp)

import time

def measure(func, *args):
    """–ò–∑–º–µ—Ä—è–µ–º –≤—Ä–µ–º—è –∏ –ø–æ—Ç—Ä–µ–±–ª–µ–Ω–∏–µ –ø–∞–º—è—Ç–∏ —Ñ—É–Ω–∫—Ü–∏–µ–π func."""
    tracemalloc.start()
    start = time.perf_counter()

    result = func(*args)

    elapsed = time.perf_counter() - start
    current, peak = tracemalloc.get_traced_memory()
    tracemalloc.stop()

    return result, elapsed, peak

def fractional_knapsack(weights, values, capacity):
    """–†–µ–∞–ª–∏–∑—É–µ—Ç –∂–∞–¥–Ω—ã–π –∞–ª–≥–æ—Ä–∏—Ç–º –¥–ª—è –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω–æ–≥–æ —Ä—é–∫–∑–∞–∫–∞."""
    items = list(zip(weights, values))
    # —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ —É–±—ã–≤–∞–Ω–∏—é value/weight
    items.sort(key=lambda x: x[1] / x[0], reverse=True)

    total_value = 0
    remaining = capacity

    for w, v in items:
        if w <= remaining:
            # –±–µ—Ä—ë–º –≤–µ—Å—å –ø—Ä–µ–¥–º–µ—Ç
            total_value += v
            remaining -= w
        else:
            # –±–µ—Ä—ë–º –¥—Ä–æ–±—å –ø—Ä–µ–¥–º–µ—Ç–∞
            total_value += v * (remaining / w)
            break

    return total_value
